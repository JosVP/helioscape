─────────────────────────────────────────────────────────────────────────────
File: src/app/features/mercury/mercury-hud/building-info/building-info.component.ts
─────────────────────────────────────────────────────────────────────────────

Selector: app-mercury-building-info
Standalone, OnPush.
Inject: GameStateService, DataService.

Inputs:
  buildingId = input.required<string>()   // PlacedBuilding.buildingId
  placedId   = input.required<string>()   // PlacedBuilding.id (uuid)
  slotId     = input<string | null>(null) // MercurySlot.id if tile has a slot

Outputs:
  closed = output<void>()

Computed:
  readonly building = computed(() => this.data.getMercuryBuilding(this.buildingId()))
  readonly placed = computed(() =>
    this.gameState.mercuryBuildings().find(b => b.id === this.placedId()))
  readonly isRefinery = computed(() =>
    this.data.getMercurySlot(this.slotId() ?? '')?.slotType === 'refinery')
  readonly miningInfo = computed(() => {
    // If this is a refinery slot, find any mining_location that lists this slotId in adjacentRefinerySlots
    const slot = this.slotId();
    if (!slot) return null;
    return this.data.getMercuryMapData()?.miningLocations
      .find(m => m.adjacentRefinerySlots.includes(slot)) ?? null;
  })
  readonly assignedMiners = computed(() => {
    // NOTE: depends on mercuryMiners signal (Block 9.6) — guard with nullish coalescing
    const miners = (this.gameState as any).mercuryMiners?.();
    return miners?.assignments?.[this.slotId() ?? ''] ?? 0;
  })

Template:

    <div class="building-info__header">
      <span class="building-info__name">{{ building()?.displayName }}</span>
      <button class="building-info__close" (click)="closed.emit()" aria-label="Close">×</button>
    </div>

    @if (placed(); as p) {
      @if (p.status === 'building') {
        <div class="building-info__status building-info__status--building">
          Under construction — {{ (p.buildProgressYears / p.totalBuildYears * 100) | number:'1.0-0' }}%

      } @else {

      }
    }

    @if (building(); as b) {
      <div class="building-info__desc">{{ b.description }}</div>
      @if (b.energyDrawGw !== 0) {
        <div class="building-info__power">
          Power: {{ b.energyDrawGw < 0 ? '+' + (-b.energyDrawGw) : b.energyDrawGw }} GW

      @for (eff of b.effects; track eff.type) {

    @if (isRefinery() && miningInfo(); as mine) {

    @if (isRefinery()) {
      <div class="building-info__miners">
        <div class="building-info__miners-label">
