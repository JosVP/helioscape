Template:
  <div class="mercury-sidebar">
    <div class="mercury-sidebar__minimap">
      <!-- Mini-map canvas: Block 9.3b (post-playtest) — render placeholder div for now -->
      <div class="mercury-sidebar__minimap-placeholder" aria-label="Mini-map coming soon"></div>
    </div>

    <div class="mercury-sidebar__tabs" role="tablist">
      @for (tab of tabs; track tab.id) {
        <button role="tab" [attr.aria-selected]="activeTab() === tab.id"
                (click)="activeTab.set(tab.id)" class="mercury-sidebar__tab"
                [class.mercury-sidebar__tab--active]="activeTab() === tab.id">
          {{ tab.label }}
        </button>
      }
    </div>

    <div class="mercury-sidebar__list" role="tabpanel">
      @if (filteredBuildings().length === 0) {
        <p class="mercury-sidebar__empty">Nothing available yet.</p>
      }
      @for (b of filteredBuildings(); track b.id) {
        <div class="mercury-sidebar__card"
             [class.mercury-sidebar__card--selected]="selectedBuildingId() === b.id"
             (click)="selectBuilding(b.id)"
             role="button" tabindex="0"
             [attr.aria-pressed]="selectedBuildingId() === b.id"
             (keydown.enter)="selectBuilding(b.id)"
             (keydown.space)="$event.preventDefault(); selectBuilding(b.id)">
          <div class="mercury-sidebar__card-name">{{ b.displayName }}</div>
          <div class="mercury-sidebar__card-cost">
            @if (b.cost.commonOre > 0) { <span>Ore: {{ b.cost.commonOre }}</span> }
            @if (b.cost.rareMetals > 0) { <span>Metals: {{ b.cost.rareMetals }}</span> }
            @if (b.cost.polarVolatiles > 0) { <span>Vol: {{ b.cost.polarVolatiles }}</span> }
          </div>
          <div class="mercury-sidebar__card-meta">
            {{ b.buildTimeYears }}yr
            @if (b.energyDrawGw < 0) { · +{{ -b.energyDrawGw }} GW }
            @else if (b.energyDrawGw > 0) { · {{ b.energyDrawGw }} GW draw }
          </div>
          <div class="mercury-sidebar__card-desc">{{ b.description }}</div>
        </div>
      }
    </div>
  </div>
